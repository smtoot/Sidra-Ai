# Semi-Structured Booking UI Flow (Smart Pack)

## Overview
This document provides the complete UI/UX design for the "Smart Pack" semi-structured booking system - a hybrid approach that balances teacher predictability with parent flexibility through recurring patterns + floating sessions.

**Target Package**: 10-session package with 10% discount
**Core Model**: 8 sessions via recurring pattern + 2 floating sessions
**Duration**: 6 weeks total
**Reschedules**: 2 per session with 24h notice requirement

```typescript
// Example calculation
function calculateExpiryDate(package: SmartPackage): Date {
  const lastScheduledSession = package.scheduledSessions[7].date; // 8th session (0-indexed)
  const gracePeriod = addWeeks(lastScheduledSession, 2);
  const maxDuration = addWeeks(package.purchaseDate, 6);
```

### Cron Jobs
```typescript
// apps/api/src/package/package-expiry.cron.ts

@Injectable()
export class PackageExpiryCron {
  @Cron('0 0 * * *') // Daily at midnight
  async checkExpiredPackages() {
    const expiredPackages = await this.prisma.package.findMany({
      where: {
        expiryDate: { lte: new Date() },
        status: 'ACTIVE',
      },
      include: {
        sessions: {
          where: { status: 'UNBOOKED' },
        },
      },
    });

    for (const pkg of expiredPackages) {
      // Count unused sessions
      const unusedCount = pkg.sessions.filter(s => s.status === 'UNBOOKED').length;

      if (unusedCount > 0) {
        // Calculate refund
        const pricePerSession = pkg.finalPrice / 10;
        const refundAmount = pricePerSession * unusedCount;

        // Process refund (atomic transaction)
        await this.processRefund(pkg.id, refundAmount);
      }

      // Update package status
      await this.prisma.package.update({
        where: { id: pkg.id },
        data: { status: 'EXPIRED' },
      });

      // Send notification
      await this.notificationService.sendPackageExpired(pkg);
    }
  }

  @Cron('0 9 * * *') // Daily at 9 AM
  async sendExpiryWarnings() {
    const packagesExpiringSoon = await this.prisma.package.findMany({
      where: {
        expiryDate: {
          gte: new Date(),
          lte: addDays(new Date(), 7), // Next 7 days
        },
        status: 'ACTIVE',
      },
      include: {
        sessions: {
          where: { status: 'UNBOOKED' },
        },
      },
    });

    for (const pkg of packagesExpiringSoon) {
      const unbookedFloating = pkg.sessions.filter(s => s.isFloating && s.status === 'UNBOOKED');

      if (unbookedFloating.length > 0) {
        const daysRemaining = differenceInDays(pkg.expiryDate, new Date());
        await this.notificationService.sendExpiryWarning(pkg, daysRemaining);
      }
    }
  }
}
```
